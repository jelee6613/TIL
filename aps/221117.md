```python
import sys
input = sys.stdin.readline

N, M = map(int, input().split())
nums = sorted(list(map(int, input().split())))

results = []

def function(arr, i):
    if i == M:
        results.append(arr)
        return

    for j in nums:
        if i == 0:
            function(arr + [j], i + 1)
        elif j >= arr[i - 1]:
            function(arr + [j], i + 1)


function([], 0)

for result in results:
    print(*result)
```
