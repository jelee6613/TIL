```python
nums = list(map(int, input()))

if nums[0] == 0:
    print(0)
else:

    if len(nums) == 1:
        print(1)
    elif len(nums) == 2:
        tmp = nums[0] * 10 + nums[1]
        if tmp == 10 or tmp == 20:
            print(1)
        elif 26 >= tmp > 10:
            print(2)
        elif nums[1] == 0:
            print(0)
        else:
            print(1)
    else:
        dp = [0] * len(nums)
        dp[0] = 1

        tmp = nums[0] * 10 + nums[1]
        if tmp == 10 or tmp == 20:
            dp[1] = 1
        elif 26 >= tmp > 10:
            dp[1] = 2
        elif nums[1] == 0:
            dp[1] = 0
            print(0)
        else:
            dp[1] = 1

        if dp[1]:

            for i in range(2, len(nums)):

                tmp = nums[i-1] * 10 + nums[i]

                if tmp == 10 or tmp == 20:
                    dp[i] = dp[i-2]
                elif 26 >= tmp > 10:
                    dp[i] = dp[i-1] + dp[i-2]
                elif not nums[i]:
                    dp[-1] = 0
                    break
                else:
                    dp[i] = dp[i-1]

            print(dp[-1] % 1000000)
```
